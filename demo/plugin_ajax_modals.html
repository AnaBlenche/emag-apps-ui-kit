AJAX Modals

eMAG Apps UI KIT - AJAX Modals

Works great with the standard Bootstrap version

                                            
var modalTest = new PhotonModal({
    id: 'testModal',
    title: 'This is an AJAX modal',
    useAjax: true,
    ajaxModalParams: {
        method: 'POST',
        url: '/emag-apps-ui-kit/demo/plugin_ajax_modals_modal_content.html'
    },
    size: 'default',
    type: 'preview',
    buttons: {
        save: {
            label: 'Save',
            class: 'btn-success',
            closeModal: true
        },
        cancel: {
            label: 'Cancel',
            class: 'btn-default',
            closeModal: true
        }
    }
});
